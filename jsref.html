let raqamlar = [42, 13, 78];
raqamlar[0];  // 42
raqamlar[2];  // 78
raqamlar.length;  // 3

raqamlar.push(99);
// raqamlar = [42, 13, 78, 99]

let oxirgi = raqamlar.pop();
// oxirgi = 99 | raqamlar = [42, 13, 78]

let boshidagi = raqamlar.shift();
// boshidagi = 42 | raqamlar = [13, 78]

raqamlar.unshift(55);
// raqamlar = [55, 13, 78]
